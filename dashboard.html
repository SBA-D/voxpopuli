async function solicitarSaque() { const valorSaque = parseFloat(document.getElementById('input-valor-saque').value); const pix = document.getElementById('input-pix-saque').value; const usuarioId = localStorage.getItem('vox_usuario'); if (!valorSaque || valorSaque <= 0) { alert("Por favor, insira um valor válido para o saque."); return; } if (!pix) { alert("Por favor, informe sua chave PIX."); return; } try { // 1. Busca o saldo atual do membro no banco const { data: membro, error: fetchError } = await clientSupabase .from('membros') .select('saldo') .eq('id', usuarioId) .single(); if (fetchError || !membro) { alert("Erro ao identificar usuário ou saldo."); return; } // 2. Verifica se o saldo é suficiente if (membro.saldo < valorSaque) { alert("Saldo insuficiente para este saque."); return; } // 3. Registra o pedido de saque na tabela 'saques' const { error: insertError } = await clientSupabase.from('saques').insert([ { membro_id: usuarioId, moedas_sacar: valorSaque, chave_pix: pix, status: 'PENDENTE' } ]); if (insertError) throw insertError; // 4. ATUALIZAÇÃO DO SALDO: Deduz o saldo imediatamente após o pedido const novoSaldo = membro.saldo - valorSaque; const { error: updateError } = await clientSupabase .from('membros') .update({ saldo: novoSaldo }) .eq('id', usuarioId); if (updateError) throw updateError; alert("Saque solicitado com sucesso! Seu saldo foi atualizado."); location.reload(); // Recarrega para mostrar o saldo novo } catch (err) { console.error("Erro ao solicitar saque:", err); alert("Ocorreu um erro ao processar seu saque. Tente novamente."); } }
