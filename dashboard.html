async function solicitarSaque() { const valorSaque = parseFloat(document.getElementById('input-valor-saque').value); const pix = document.getElementById('input-pix-saque').value; const usuarioId = String(localStorage.getItem('vox_usuario')).trim(); if (!valorSaque || isNaN(valorSaque) || valorSaque <= 0) { alert("Valor inválido."); return; } try { // 1. Busca o membro const { data: membro, error: fetchError } = await clientSupabase .from('membros') .select('id, saldo') .eq('id', usuarioId) .maybeSingle(); if (fetchError || !membro) { alert("Erro: Usuário não localizado no sistema."); return; } const saldoAtual = parseFloat(membro.saldo || 0); if (saldoAtual < valorSaque) { alert("Saldo insuficiente."); return; } // 2. Insere pedido de saque await clientSupabase.from('saques').insert([ { membro_id: usuarioId, moedas_sacar: valorSaque, chave_pix: pix, status: 'PENDENTE' } ]); // 3. ATUALIZAÇÃO FORÇADA // Se a coluna no banco for 'saldo', o comando abaixo deve funcionar. // Se você não tiver certeza absoluta do nome da coluna, verifique no painel do Supabase. const { error: updateError } = await clientSupabase .from('membros') .update({ saldo: (saldoAtual - valorSaque) }) .match({ id: usuarioId }); // Usamos .match para garantir segurança na seleção if (updateError) { console.error("Erro no update:", updateError); alert("Erro ao debitar saldo. Veja o console F12."); return; } alert("Sucesso! Saque solicitado."); location.reload(); } catch (err) { console.error(err); } }
