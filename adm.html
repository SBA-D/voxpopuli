async function liquidarEnqueteComRateio(enqueteId, opcaoVencedora, nomeAtivo) { if(!confirm(`Confirmar vitória de "${opcaoVencedora}" com cálculo de rateio?`)) return; // 1. Pega todas as apostas que foram feitas nessa enquete específica let { data: todasAsApostas } = await _supabase.from('apostas').select('*').eq('enquete_id', enqueteId); if(todasAsApostas && todasAsApostas.length > 0) { let poteTotal = 0; let poteVencedores = 0; let vencedores = []; // 2. Soma o dinheiro total da enquete (Pote Total) // e descobre quanto os acertadores apostaram juntos (Pote Vencedores) todasAsApostas.forEach(aposta => { poteTotal += parseFloat(aposta.valor); if(aposta.palpite === opcaoVencedora) { poteVencedores += parseFloat(aposta.valor); vencedores.push(aposta); // Guarda quem acertou } }); // 3. Se houver ganhadores, faz a divisão proporcional (o rateio) if(vencedores.length > 0 && poteVencedores > 0) { for(let vencedor of vencedores) { // Descobre a porcentagem que este membro representa no grupo dos vencedores let proporcao = parseFloat(vencedor.valor) / poteVencedores; // O prêmio dele é a parte proporcional do Pote Total arrecadado! let premioGanho = poteTotal * proporcao; // Busca o saldo atual dele no banco para não ter erro let { data: u } = await _supabase.from('usuarios').select('saldo').eq('username', vencedor.usuario).single(); let saldoNovo = parseFloat(u.saldo || 0) + premioGanho; // 4. ATUALIZA O SALDO DELE (SOMA O LUCRO) E CRIA O HISTÓRICO await _supabase.from('usuarios').update({ saldo: saldoNovo }).eq('username', vencedor.usuario); await _supabase.from('transacoes').insert([{ usuario: vencedor.usuario, tipo: 'premio', descricao: `Prêmio/Rateio ganho na enquete [${nomeAtivo}]`, valor: premioGanho // Valor positivo que vai aparecer no extrato }]); } } } // 5. Finaliza a enquete para sumir do painel deles await _supabase.from('enquetes').update({ finalizada: true }).eq('id', enqueteId); alert("Mercado liquidado e saldo distribuído via rateio!"); inicializarPainel(); }
